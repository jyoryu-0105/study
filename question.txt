# [i]でインサートモードに入ります
# インサートモードから抜け出すときは [Esc]
# なお、下記コマンドをこの文章内では「正しいコマンド」と呼びます
echo Hello World

# 手順通りに進んでいき、全て終わったら下記を実行してください
# 変更内容を保存してVimを終了する [:wq]

# 参考：その他終了するときのコマンド
# Vimを終了する（ファイルの中身を編集している場合はできません） [:q]
# 変更内容を保存しないでVimを強制終了する [:q!]
# 変更内容を保存 [:w]

# では下記を実行してください

# 行末に移動 [$]
# インサートモードで文末にdを追加して正しいコマンドにしてください
# 終了後はインサートモードから抜け出す [Esc]
echo Hello Worl

# 行頭に移動 [0]
# インサートモードで先頭にeを追加して正しいコマンドにしてください
# 終了後はインサートモードから抜け出す [Esc]
cho Hello World

# 単語ごとに前方移動したい時は[w]
# 単語ごとに後方移動したい時は[b]
# インサートモードで正しいコマンドにしてください
echo ello World

# 行をヤンク(コピー)したい [yy]
# 行を削除したい [dd]
# 行をペーストしたい [p]
# この下のコマンドを削除し、正しいコマンド(4行目)をヤンクしてペーストしてください
# [数字 G]で指定の行数まで戻れます。
# ちなみに下のコマンドの行数は37行目です。
echo Go To Hell

# 文字列を検索する[/]
# 一旦インサートモードを終了してから[txt]と入力してください
# その後[n]を押すと検索行った後に、次の検索結果に移動します






























































# ここでインサートモードに入り、正しいコマンドにしてください
echo Hello txt

# 1文字だけデリート [x]
echo Hello Worldd

# 行頭までデリート [d0]
# 先に行末まで移動してください [$]
echo Hello Wood

# 行末までデリート [d$]
# 先に行頭まで移動してください [0]
echo Hello Wood

# 単語をひとつづつデリート [dw]
echo Hello Wood

# 最後の行までデリート [dG]
# 終わったら[gg]で先頭に戻ってください。
echo Hello Wood
echo Hello Wood
echo Hello Wood
echo Hello Wood
echo Hello Wood
